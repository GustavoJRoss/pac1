Condicionais em C

Introdução

As estruturas condicionais permitem que um programa tome decisões baseadas em condições específicas. Em C, as principais estruturas condicionais são if, if-else, else-if e switch. Estas estruturas ajudam a controlar o fluxo de execução do programa, permitindo que certas partes do código sejam executadas somente quando determinadas condições são atendidas.

Estrutura if

A estrutura if é usada para executar um bloco de código se uma condição específica for verdadeira. Se a condição for falsa, o bloco de código é ignorado.

#include <stdio.h>

int main() {
    int idade = 18;
    
    if (idade >= 18) {
        printf("Você é maior de idade.\n");
    }
    
    return 0;
}

Estrutura if-else

A estrutura if-else é uma extensão da estrutura if. Ela permite executar um bloco de código se a condição for verdadeira e outro bloco se a condição for falsa.

#include <stdio.h>

int main() {
    int idade = 16;
    
    if (idade >= 18) {
        printf("Você é maior de idade.\n");
    } else {
        printf("Você é menor de idade.\n");
    }
    
    return 0;
}

Estrutura else-if

A estrutura else-if permite testar múltiplas condições. Se a primeira condição for falsa, a próxima condição é testada, e assim por diante, até que uma condição verdadeira seja encontrada ou que todas as condições sejam testadas.

#include <stdio.h>

int main() {
    int nota = 75;
    
    if (nota >= 90) {
        printf("Nota A.\n");
    } else if (nota >= 80) {
        printf("Nota B.\n");
    } else if (nota >= 70) {
        printf("Nota C.\n");
    } else if (nota >= 60) {
        printf("Nota D.\n");
    } else {
        printf("Nota F.\n");
    }
    
    return 0;
}

Estrutura switch

A estrutura switch é usada para testar a igualdade de uma variável contra uma lista de valores. Cada valor é chamado de "case". Se a variável coincidir com um case, o bloco de código correspondente é executado.

#include <stdio.h>

int main() {
    int dia = 3;
    
    switch (dia) {
        case 1:
            printf("Domingo.\n");
            break;
        case 2:
            printf("Segunda-feira.\n");
            break;
        case 3:
            printf("Terça-feira.\n");
            break;
        case 4:
            printf("Quarta-feira.\n");
            break;
        case 5:
            printf("Quinta-feira.\n");
            break;
        case 6:
            printf("Sexta-feira.\n");
            break;
        case 7:
            printf("Sábado.\n");
            break;
        default:
            printf("Dia inválido.\n");
    }
    
    return 0;
}

Exemplos de Código

Exemplo com if e if-else

#include <stdio.h>

int main() {
    int idade;
    
    printf("Digite sua idade: ");
    scanf("%d", &idade);
    
    if (idade >= 18) {
        printf("Você é maior de idade.\n");
    } else {
        printf("Você é menor de idade.\n");
    }
    
    return 0;
}

Exemplo com else-if

#include <stdio.h>

int main() {
    float temperatura;
    
    printf("Digite a temperatura em graus Celsius: ");
    scanf("%f", &temperatura);
    
    if (temperatura >= 30) {
        printf("Está muito quente.\n");
    } else if (temperatura >= 20) {
        printf("O tempo está agradável.\n");
    } else if (temperatura >= 10) {
        printf("Está frio.\n");
    } else {
        printf("Está muito frio.\n");
    }
    
    return 0;
}

Exemplo com switch

#include <stdio.h>

int main() {
    int opcao;
    
    printf("Escolha uma opção (1-3): ");
    scanf("%d", &opcao);
    
    switch (opcao) {
        case 1:
            printf("Você escolheu a opção 1.\n");
            break;
        case 2:
            printf("Você escolheu a opção 2.\n");
            break;
        case 3:
            printf("Você escolheu a opção 3.\n");
            break;
        default:
            printf("Opção inválida.\n");
    }
    
    return 0;
}
